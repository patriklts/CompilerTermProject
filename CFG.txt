CODE -> TYPE CODE  
CODE -> ''
TYPE -> vtype CODE'
CODE' -> VDECL
CODE' -> FDECL
VDECL -> id VDECL'
VDECL' -> semi 
VDECL' -> ASSIGN semi
ASSIGN -> assign RHS
RHS -> EXPR 
RHS -> literal 
RHS -> character 
RHS -> boolstr
EXPR -> lparen EXPR rparen EXPR'  
EXPR -> num EXPR'
EXPR -> id EXPR' 
EXPR' -> addsub EXPR EXPR'
EXPR' -> multdiv EXPR EXPR'
EXPR' -> ''
FDECL -> id lparen ARG rparen lbrace BLOCK RETURN rbrace
ARG -> vtype id MOREARGS 
ARG -> ''
MOREARGS -> comma vtype id MOREARGS 
MOREARGS -> ''
BLOCK -> STMT BLOCK 
BLOCK -> ''
STMT -> VDECL 
STMT -> ASSIGN semi
STMT -> if lparen COND rparen lbrace BLOCK rbrace ELSE
STMT -> while lparen COND rparen lbrace BLOCK rbrace
COND -> boolstr COND'
COND' -> comp COND 
COND' -> ''
ELSE -> else lbrace BLOCK rbrace 
ELSE -> ''
RETURN -> return RHS semi
